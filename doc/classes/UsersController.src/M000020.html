  def destroy
    @user = User.find(params[:id])
    begin
            @user.destroy
            flash[:notice] = "User #{@user.name} deleted"      
    rescue Exception => e
            flash[:error] = e.message
    end
    
    respond_to do |format|
      format.html { redirect_to(users_url) }
      format.xml  { head :ok }
    end
  end
